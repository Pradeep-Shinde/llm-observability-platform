openwebui/backend/open_webui/main.py
    from open_webui.routers import metrics, ai_observability

    app.include_router(metrics.router, prefix="/metrics", tags=["metrics"])

    from open_webui.metrics.middleware import MetricsMiddleware

    app.add_middleware(MetricsMiddleware)

    app.include_router(ai_observability.router, prefix="/api/ai", tags=["AI Observability"])

deploy service:
cd open-webui/backend/
python3 -m venv venv-dev
source venv-dev/bin/activate
pip install -r requirements.txt

install required libraries like
python3 -m pip install prometheus-client
python3 -m pip install psutil
python3 -m pip install openai

export ENABLE_SIGNUP=true
export REDIS_URL=redis://:<redis pass>@<redis user>:<port>

for sql lite db
export DATABASE_URL=sqlite:///./dev.db
or
export DATABASE_URL=postgresql://postgres:<pass>@<user>:<port>/postgres

export LITELLM_URL =http://localhost:4000/v1
export PROM_URL =http://localhost:19090

python -m uvicorn open_webui.main:app --host 0.0.0.0 --port 3000

also check newly created <open-webui-url>/metrics/ endpoint which shows all the metrics
